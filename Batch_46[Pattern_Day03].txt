int main() {
  int n;
  printf("Enter n value : ");
  scanf("%d",&n);
  int num = 2;
  for(int i = 1;i<=n;i++) {
   int count = n;
   while(count!=0) {
       int c = 0;
       for(int j = 2;j<=num/2;j++) {
           if(num%j==0) {
               c++;
           }
       }
       if(c==0) {
           printf("%3d",num);
           count--;
       }
       num++;
   }
      printf("\n");
  }
    

    return 0;
}
